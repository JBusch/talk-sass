Mixins!

@mixin name ($variable, ...) {eigenschaften}

@include name ($variable, ...)

Mixins!

mixin sexy-border($color, $width) {
  border: {
    color: $color;
    width: $width;
    style: dashed;
  }
}


p { @include sexy-border(blue, 1px); }
kompiliert zu 

p {
  border-color: blue;
  border-width: 1px;
  border-style: dashed;
}
Mixins!

.news-list-container {
  @include grid-row();

  .news-list-item {
    @include grid-column(12);

    @media #{$medium-only} { 
      @include grid-column(6);
    }
    
  }
}
