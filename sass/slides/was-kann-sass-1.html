Vorsicht!

@extend ist mit Vorsicht zu genießen!


.important {
  font-weight: bold;
}
 
.sidebar .important {
  color: red;
}
 
.message {
  @extend .important;
}
kompiliert zu 

.important, .message {
  font-weight: bold;
}
 
.sidebar .important, .sidebar .message {
  color: red;
}

.clearfix {
  &:before,
  &:after {
    content: " ";
    display: table;
  }
  &:after {
    clear: both;
  }
}

.module {
  padding: 10px;
  @extend .clearfix;
  h3 {
    color: red;
    @extend .clearfix;
    span {
      float: right; 
    }
  }
}

.sports {
  @extend .module;
}

.news {
  @extend .module; 
}
kompiliert zu 

.clearfix:before,
.module:before,
.sports:before,
.news:before,
.module h3:before,
.sports h3:before,
.news h3:before,
.clearfix:after,
.module:after,
.sports:after,
.news:after,
.module h3:after,
.sports h3:after,
.news h3:after {
  content: " ";
  display: table;
}
.clearfix:after,
.module:after,
.sports:after,
.news:after,
.module h3:after,
.sports h3:after,
.news h3:after {
  clear: both;
}

.module, .sports, .news {
  padding: 10px;
}
.module h3, .sports h3, .news h3 {
  color: red;
}
.module h3 span, .sports h3 span, .news h3 span {
  float: right;
}
